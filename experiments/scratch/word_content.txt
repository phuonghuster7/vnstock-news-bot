symbol → {entry_price, stop_price, entry_date}        def add_position(self, symbol: str, entry_price: float,                      high_series: pd.Series, low_series: pd.Series,                      close_series: pd.Series):        atr = self._calc_atr(high_series, low_series, close_series)        stop = entry_price - self.atr_mult * atr        self.positions[symbol] = {            "entry": entry_price,            "stop":  round(stop, 2),        }        def _calc_atr(self, high, low, close):        tr = pd.concat([            high - low,            (high - close.shift(1)).abs(),            (low  - close.shift(1)).abs(),        ], axis=1).max(axis=1)        return tr.rolling(self.atr_window).mean().iloc[-1]        def check_stops(self, current_prices: dict) -> list[str]:        """Trả về list symbols cần sell vì chạm stop."""        triggered = []        for sym, pos in self.positions.items():            price = current_prices.get(sym)            if price and price <= pos["stop"]:                triggered.append(sym)        return triggered        def remove_position(self, symbol: str):        self.positions.pop(symbol, None)Bước 3 — Fundamental đúng cách: dùng làm filter, không phải featureThay vì cho LightGBM học từ fundamental → dùng fundamental như pre-filter trước khi chạy signal:# experiments/fundamental_filter.pyclass FundamentalFilter:    """    Loại bỏ cổ phiếu có fundamental xấu TRƯỚC khi chạy momentum signal.    Không để LightGBM quyết định — rule-based rõ ràng hơn.        Universe VN30 → filter → ~20 symbols → momentum rank → top 10    """        RULES = {        "pe_max":    40.0,   #
